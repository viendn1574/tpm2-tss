19 #define CHECK_IN_LIST(type, needle, ...) \
20  type tab[] = { __VA_ARGS__ }; \
21  size_t i; \
22  for(i = 0; i < sizeof(tab) / sizeof(tab[0]); i++) \
23  if (needle == tab[i]) \
24  break; \
25  if (i == sizeof(tab) / sizeof(tab[0])) { \
26  LOG_ERROR("Bad value"); \
27  return TSS2_FAPI_RC_BAD_VALUE; \
28  }
